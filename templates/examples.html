import requests
import json
baseurl = 'https://plataformaiotalfa.herokuapp.com/'
authtoken = 'your token'
headers = {'Content-Type': 'application/json', 'Authorization': f'Token {authtoken}'}
serial = 'serial do dispositivo'

Obter o dados do dispositivo
url = f'{baseurl}/api/v1/dispositivos/{serial}/'
request = requests.get(headers=headers url=url)
response = request.json()
dados = response[0]['dados']
Obter limites do dispositivo
url = f'{baseurl}/api/v1/dispositivos/{serial}/'
request = requests.get(headers=headers url=url)
response = request.json()
confs = response[0]['configuracoes'][0]
inferior = confs['limite_inferior']
superior = confs['limite_superior']
Obter mensagens do dispositivo
url = f'{baseurl}/api/v1/dispositivos/{serial}/'
request = requests.get(headers=headers url=url)
response = request.json()
mensagens = response[0]['mensagens']
